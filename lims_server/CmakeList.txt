cmake_minimum_required(VERSION 3.10)
project(lab_server)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_FLAGS "-Wall -O2")

# 查找依赖库
find_package(Poco REQUIRED COMPONENTS Net Util)
find_package(libpqxx REQUIRED)
find_package(nlohmann_json REQUIRED)

# 包含头文件目录
include_directories(include)

# 收集源文件
file(GLOB SOURCES "src/*.cpp" "src/network/*.cpp" "src/service/*.cpp" "src/dao/*.cpp" "src/common/*.cpp")

# 生成可执行文件
add_executable(lab_server main.cpp ${SOURCES})

# 链接依赖库
target_link_libraries(lab_server 
    Poco::Net 
    Poco::Util 
    pqxx 
    nlohmann_json::nlohmann_json
    pthread  # 线程库
)
